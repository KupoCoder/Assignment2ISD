DROP TABLE Order_T;
DROP TABLE Payment_T;
DROP TABLE AccessLog_T;
DROP TABLE Staff_T;
DROP TABLE Customer_T;
DROP TABLE Device_T;
DROP TABLE Admin_T;

CREATE Table Staff_T(
    staffID VARCHAR(15) NOT NULL,
    staffFirstName VARCHAR(30),
    staffLastName VARCHAR(30),
    staffEmail VARCHAR(100),
    staffPosition VARCHAR(20),
    staffUsername VARCHAR(20),
    staffPassword VARCHAR(20),
    CONSTRAINT Staff_T_PK PRIMARY KEY (staffID)
);

CREATE Table Admin_T(
    adminID VARCHAR(15) NOT NULL,
    adminEmail VARCHAR(100),
    adminPassword VARCHAR(20),
    CONSTRAINT Admin_T_PK PRIMARY KEY (adminID)
);

CREATE Table Customer_T(
    customerID VARCHAR(15) NOT NULL,
    customerFirstName VARCHAR(30),
    customerLastName VARCHAR(30),
    customerDOB DATE,
    customerEmail VARCHAR(99),
    customerPhoneNumber VARCHAR(15),
    customerStreet VARCHAR(50),
    customerSuburb VARCHAR(50),
    customerState VARCHAR(3),
    customerPostCode VARCHAR(10),
    customerUsername VARCHAR(30),
    customerPassword VARCHAR(30),
    CONSTRAINT Customer_T_PK PRIMARY KEY (customerID)
);

CREATE Table Payment_T(
    paymentID VARCHAR(15) NOT NULL,
    paymentCardName VARCHAR(30),
    paymentCardNumber BIGINT,
    paymentCardCVC BIGINT,
    paymentCardExpiryDate DATE,
    customerID VARCHAR(15),
    CONSTRAINT Payment_T_PK PRIMARY KEY (paymentID),
    CONSTRAINT Payment_T_FK FOREIGN KEY (customerID) REFERENCES Customer_T (customerID)
);

CREATE Table Order_T(
    orderID VARCHAR(15) NOT NULL,
    customerID VARCHAR(15),
    orderDate DATE,
    orderTotalAmount FLOAT,
    orderStatus VARCHAR(20),
    orderComplete BOOLEAN,
    staffID VARCHAR(15),
    paymentID VARCHAR(15),
    CONSTRAINT Order_T_PK PRIMARY KEY (orderID),
    CONSTRAINT Order_T_FK1 FOREIGN KEY (customerID) REFERENCES Customer_T (customerID),
    CONSTRAINT Order_T_FK2 FOREIGN KEY (staffID) REFERENCES Staff_T (staffID),
    CONSTRAINT Order_T_FK3 FOREIGN KEY (paymentID) REFERENCES Payment_T (paymentID)
);

CREATE Table Device_T(
    deviceID VARCHAR(15) NOT NULL,
    deviceName VARCHAR (200),
    deviceDescription VARCHAR(2000),
    deviceBrand VARCHAR(50),
    deviceSupplier VARCHAR(50),
    deviceSpecifications VARCHAR(2000),
    deviceWRPolicy VARCHAR(1000),
    deviceStandardPrice FLOAT,
    deviceCurrentPrice FLOAT,
    deviceType VARCHAR(50),
    deviceSOH INT,
    deviceImage VARCHAR(200), --link to image
    CONSTRAINT Device_T_PK PRIMARY KEY (deviceID)
);

CREATE Table OrderLine_T(
    orderlineID VARCHAR(15) NOT NULL,
    orderID VARCHAR(15),
    deviceID VARCHAR(15),
    orderlineDateAdded TIMESTAMP,
    orderlineQuantity INT,
    CONSTRAINT OrderLine_T_PK PRIMARY KEY (orderlineID),
    CONSTRAINT OrderLine_T_FK1 FOREIGN KEY (orderID) REFERENCES Order_T (orderID),
    CONSTRAINT OrderLine_T_FK2 FOREIGN KEY (deviceID) REFERENCES Device_T (deviceID)
);

CREATE Table AccessLog_T(
    logID VARCHAR(15) NOT NULL,
    customerID VARCHAR(15),
    staffID VARCHAR(15),
    logLogin TIMESTAMP,
    logLogout TIMESTAMP,
    CONSTRAINT AccessLog_T_PK PRIMARY KEY (logID),
    CONSTRAINT AccessLog_T_FK1 FOREIGN KEY (customerID) REFERENCES Customer_T (customerID),
    CONSTRAINT AccessLog_T_FK2 FOREIGN KEY (staffID) REFERENCES Staff_T (staffID)
);
